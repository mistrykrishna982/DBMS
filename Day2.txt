4.How many patients does Dr. Mehta see this week?

->select count(*) as total_patients from appointment join doctor on appointment.d_id=doctor.d_id where d_nm="Dr. Mehta";

5.Which department earns the most from consultations?

 -> select dept_nm, sum(fee) as total_income from appointment join doctor on appointment.d_id=doctor.d_id join department on doctor.dept_id=department.dept_id group by dept_nm order by total_income desc;
